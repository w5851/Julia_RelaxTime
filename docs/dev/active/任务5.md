## 一、核心要求

### 常见期刊格式要求：
- **PRL/PRD/PR系列**：
  - 宽度：单栏 3.375 英寸，双栏 6.75 英寸
  - 字体：Times New Roman，8-10pt
  - 分辨率：600 DPI（位图）或矢量格式
  - 线条粗细：≥0.5pt（约0.176mm）
  - 常用格式：PDF, EPS, PNG

## 二、完整代码示例

```python
import numpy as np
import matplotlib.pyplot as plt
from matplotlib import rcParams

# ====================
# 1. 设置PRL/PRD期刊样式
# ====================
def set_prl_style(single_column=True):
    """
    设置PRL/PRD期刊的绘图样式
    
    Parameters:
    -----------
    single_column : bool
        True: 单栏图 (3.375英寸宽)
        False: 双栏图 (6.75英寸宽)
    """
    plt.rcParams.update({
        # 字体设置
        'font.family': 'serif',
        'font.serif': ['Times New Roman', 'Times'],
        'font.size': 10,           # 正文字体大小
        'axes.titlesize': 10,      # 标题字体大小
        'axes.labelsize': 10,      # 坐标轴标签字体大小
        'legend.fontsize': 8,      # 图例字体大小
        'xtick.labelsize': 8,      # x轴刻度字体大小
        'ytick.labelsize': 8,      # y轴刻度字体大小
        
        # 图形尺寸
        'figure.figsize': (3.375, 2.5) if single_column else (6.75, 4.0),
        'figure.dpi': 600,         # 输出分辨率
        
        # 线条设置
        'lines.linewidth': 1.0,    # 线条宽度
        'lines.markersize': 4,     # 标记大小
        
        # 坐标轴设置
        'axes.linewidth': 0.5,     # 坐标轴线宽
        'axes.grid': False,        # 默认不显示网格
        'axes.labelpad': 2.0,      # 标签与坐标轴的距离
        
        # 刻度设置
        'xtick.major.width': 0.5,  # x轴主刻度线宽
        'ytick.major.width': 0.5,  # y轴主刻度线宽
        'xtick.minor.width': 0.25, # x轴次刻度线宽
        'ytick.minor.width': 0.25, # y轴次刻度线宽
        'xtick.major.size': 3.5,   # x轴主刻度长度
        'ytick.major.size': 3.5,   # y轴主刻度长度
        
        # 图例设置
        'legend.frameon': False,   # 不显示图例边框
        'legend.handlelength': 1.5,# 图例句柄长度
        
        # 保存设置
        'savefig.dpi': 600,
        'savefig.format': 'pdf',   # 推荐使用PDF或EPS
        'savefig.bbox': 'tight',
        'savefig.pad_inches': 0.05,
    })

# ====================
# 2. 创建符合期刊要求的图形
# ====================
def create_prl_figure(data_dict, single_column=True):
    """
    创建符合PRL格式的图形
    
    Parameters:
    -----------
    data_dict : dict
        包含数据的字典
    single_column : bool
        是否使用单栏宽度
    """
    # 设置样式
    set_prl_style(single_column)
    
    # 创建图形和坐标轴
    fig, ax = plt.subplots(1, 1)
    
    # 绘制数据（示例）
    x = data_dict.get('x', np.linspace(0, 10, 100))
    
    # 多条曲线示例
    colors = ['#1f77b4', '#ff7f0e', '#2ca02c', '#d62728']
    linestyles = ['-', '--', '-.', ':']
    
    for i, (label, y_data) in enumerate(data_dict.get('curves', {}).items()):
        ax.plot(x, y_data, 
                color=colors[i % len(colors)],
                linestyle=linestyles[i % len(linestyles)],
                linewidth=1.0,
                label=label)
    
    # 设置坐标轴标签（使用LaTeX格式）
    ax.set_xlabel(r'$\mathrm{Wavenumber\ k\ (m^{-1})}$', fontsize=10)
    ax.set_ylabel(r'$\mathrm{Amplitude\ A(k)}$', fontsize=10)
    
    # 设置刻度
    ax.tick_params(axis='both', which='major', direction='in', 
                   length=3, width=0.5, labelsize=8)
    ax.tick_params(axis='both', which='minor', direction='in', 
                   length=1.5, width=0.25)
    
    # 添加图例
    ax.legend(loc='best', frameon=False, fontsize=8)
    
    # 设置坐标轴范围
    ax.set_xlim([0, 10])
    # ax.set_ylim([0, 1])
    
    # 可选：添加网格
    # ax.grid(True, which='major', linestyle='--', alpha=0.3, linewidth=0.5)
    
    # 调整布局
    plt.tight_layout(pad=0.5)
    
    return fig, ax

# ====================
# 3. 保存多种格式
# ====================
def save_for_pr_journals(fig, filename_base, formats=None):
    """
    保存为期刊要求的多种格式
    
    Parameters:
    -----------
    fig : matplotlib.figure.Figure
        图形对象
    filename_base : str
        基础文件名（不包含扩展名）
    formats : list
        要保存的格式列表
    """
    if formats is None:
        formats = ['pdf', 'eps', 'png']  # 推荐的格式
    
    for fmt in formats:
        if fmt == 'eps':
            fig.savefig(f'{filename_base}.eps', format='eps', dpi=600)
        elif fmt == 'pdf':
            fig.savefig(f'{filename_base}.pdf', format='pdf', dpi=600)
        elif fmt == 'png':
            fig.savefig(f'{filename_base}.png', format='png', dpi=600, 
                       bbox_inches='tight', pad_inches=0.05)
        elif fmt == 'tiff':
            fig.savefig(f'{filename_base}.tiff', format='tiff', dpi=600)
        print(f"Saved: {filename_base}.{fmt}")

# ====================
# 4. 使用示例
# ====================
if __name__ == "__main__":
    # 生成示例数据
```

### 1. **多子图布局**
```python
def create_multipanel_figure():
    """创建多面板图形（常见于PRL）"""
    set_prl_style(single_column=True)
    
    # 创建2x2的子图
    fig, axes = plt.subplots(2, 2, figsize=(6.75, 4.5))
    axes = axes.flatten()
    
    # 每个子图的设置
    for i, ax in enumerate(axes):
        ax.plot(np.random.randn(100).cumsum())
        ax.set_xlabel(r'$\mathrm{Time}$', fontsize=9)
        ax.set_ylabel(r'$\mathrm{Value}$', fontsize=9)
        
        # 添加子图标签 (a), (b), (c), (d)
        ax.text(0.05, 0.95, f'({chr(97+i)})', 
                transform=ax.transAxes,
                fontsize=10, fontweight='bold',
                verticalalignment='top')
    
    plt.tight_layout(pad=1.0, w_pad=0.5, h_pad=0.5)
    return fig
```

### 2. **颜色映射设置（用于热图）**
```python
def create_density_plot():
    """创建密度图/热图"""
    from matplotlib.colors import LinearSegmentedColormap
    
    # 自定义颜色映射
    colors = ["#0000FF", "#FFFFFF", "#FF0000"]  # 蓝-白-红
    cmap = LinearSegmentedColormap.from_list("custom", colors, N=256)
    
    set_prl_style(single_column=True)
    fig, ax = plt.subplots()
    
    # 生成数据
    x = y = np.linspace(-3, 3, 100)
    X, Y = np.meshgrid(x, y)
    Z = np.sin(X) * np.cos(Y)
    
    # 绘制热图
    im = ax.imshow(Z, cmap=cmap, aspect='auto',
                   extent=[x.min(), x.max(), y.min(), y.max()])
    
    # 添加颜色条
    cbar = plt.colorbar(im, ax=ax, pad=0.02)
    cbar.ax.tick_params(labelsize=8)
    cbar.set_label(r'$\mathrm{Intensity}$', fontsize=9)
    
    ax.set_xlabel(r'$x$', fontsize=10)
    ax.set_ylabel(r'$y$', fontsize=10)
    
    plt.tight_layout()
    return fig
```

```python
def batch_process_figures(data_list, output_dir='figures'):
    """批量生成多个图形"""
    import os
    os.makedirs(output_dir, exist_ok=True)
    
    for i, data in enumerate(data_list):
        fig, _ = create_prl_figure(data)
        fig.savefig(f'{output_dir}/figure_{i+1:02d}.pdf',
                   bbox_inches='tight')
        plt.close(fig)  # 关闭图形以释放内存
```
